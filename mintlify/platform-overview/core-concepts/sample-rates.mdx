export const HISTORICAL_KEYS = ['USD-MXN', 'USD-INR', 'USD-PHP', 'USD-NGN', 'EUR-USD', 'GBP-INR'];

export const formatNumber = (num, decimals) => {
  return num.toLocaleString('en-US', {
    minimumFractionDigits: decimals,
    maximumFractionDigits: decimals,
  });
};

export const getRateDecimals = (rate) => {
  if (rate >= 1000) return 0;
  if (rate >= 100) return 2;
  if (rate >= 1) return 3;
  return 4;
};

export const getReceiveDecimals = (rate) => {
  if (rate >= 100) return 0;
  return 2;
};

export const RateExplorer = () => {
  const [sourceCurrency, setSourceCurrency] = React.useState('USD');
  const [amount, setAmount] = React.useState(10000);
  const [historicalCorridor, setHistoricalCorridor] = React.useState('USD-MXN');
  const [selectedCodes, setSelectedCodes] = React.useState(POPULAR_CODES);
  const [searchQuery, setSearchQuery] = React.useState('');

  const sourceData = CURRENCIES.find(c => c.code === sourceCurrency);
  const destinations = CURRENCIES.filter(c => c.code !== sourceCurrency);
  const activeCodes = selectedCodes.filter(c => c !== sourceCurrency);

  const allCorridors = destinations.map(dest => {
    const midRate = dest.usdMid / sourceData.usdMid;
    const gridRate = midRate * (1 - dest.gridSpread / 100);
    const wiseRate = midRate * (1 - dest.wiseSpread / 100);
    const bankRate = midRate * (1 - dest.bankMarkup / 100);
    const gridReceive = amount * gridRate;
    const wiseReceive = amount * wiseRate;
    const bankReceive = amount * bankRate;
    const advantage = gridReceive - bankReceive;
    const advantagePct = bankReceive > 0 ? (advantage / bankReceive) * 100 : 0;
    return {
      code: dest.code, flag: dest.flag, name: dest.name,
      midRate, gridRate, wiseRate, bankRate,
      gridReceive, wiseReceive, bankReceive,
      advantage, advantagePct,
    };
  });

  const corridors = allCorridors.filter(c => activeCodes.includes(c.code));

  const avgGridPct = corridors.length > 0
    ? corridors.reduce((s, c) => s + (c.gridRate / c.midRate) * 100, 0) / corridors.length : 0;
  const avgWisePct = corridors.length > 0
    ? corridors.reduce((s, c) => s + (c.wiseRate / c.midRate) * 100, 0) / corridors.length : 0;
  const avgBankPct = corridors.length > 0
    ? corridors.reduce((s, c) => s + (c.bankRate / c.midRate) * 100, 0) / corridors.length : 0;
  const avgSavingsInSource = corridors.length > 0
    ? corridors.reduce((s, c) => s + (c.advantage / c.midRate), 0) / corridors.length : 0;

  const isCustomized = selectedCodes.length !== POPULAR_CODES.length ||
    !POPULAR_CODES.every(c => selectedCodes.includes(c));

  const addableCurrencies = destinations.filter(c => !selectedCodes.includes(c.code));
  const searchMatches = searchQuery.length > 0 ? addableCurrencies.filter(c =>
    c.code.toLowerCase().includes(searchQuery.toLowerCase()) ||
    c.name.toLowerCase().includes(searchQuery.toLowerCase())
  ).slice(0, 8) : [];

  const addCurrency = (code) => {
    setSelectedCodes([...selectedCodes, code]);
    setSearchQuery('');
  };

  const removeCurrency = (code) => {
    setSelectedCodes(selectedCodes.filter(c => c !== code));
  };

  const showAll = () => {
    setSelectedCodes(CURRENCIES.map(c => c.code));
  };

  const resetSelection = () => {
    setSelectedCodes(POPULAR_CODES);
  };

  const histData = HISTORICAL_DATA[historicalCorridor];
  const histDest = histData ? histData.dest : '';

  return (
    <div>
      <div className="rate-explorer not-prose">
        <div className="rate-explorer-controls">
          <div className="rate-explorer-field">
            <label className="rate-explorer-label">You send</label>
            <select
              className="rate-explorer-select"
              value={sourceCurrency}
              onChange={(e) => setSourceCurrency(e.target.value)}
            >
              {SOURCE_OPTIONS.map((opt) => (
                <option key={opt.value} value={opt.value}>
                  {opt.flag} {opt.label}
                </option>
              ))}
            </select>
          </div>
          <div className="rate-explorer-field">
            <label className="rate-explorer-label">Amount</label>
            <div className="rate-explorer-input-wrapper">
              <span className="rate-explorer-input-prefix">{sourceCurrency}</span>
              <input
                type="number"
                className="rate-explorer-input"
                value={amount}
                min={1}
                onChange={(e) => {
                  const val = parseFloat(e.target.value);
                  setAmount(val > 0 ? val : 1);
                }}
              />
            </div>
          </div>
        </div>

        <div className="rate-explorer-summary">
          <span className="rate-explorer-summary-stat">
            <strong>Grid</strong> delivers <strong>{avgGridPct.toFixed(1)}%</strong> of mid-market
          </span>
          <span className="rate-explorer-summary-dot">{'\u00b7'}</span>
          <span className="rate-explorer-summary-stat">Wise: {avgWisePct.toFixed(1)}%</span>
          <span className="rate-explorer-summary-dot">{'\u00b7'}</span>
          <span className="rate-explorer-summary-stat">Banks: ~{avgBankPct.toFixed(0)}%</span>
          <span className="rate-explorer-summary-dot">{'\u00b7'}</span>
          <span className="rate-explorer-summary-stat">
            On {sourceCurrency} {formatNumber(amount, 0)} that{'\u2019'}s ~{sourceCurrency} {formatNumber(Math.round(avgSavingsInSource), 0)} more per transfer
          </span>
        </div>

        <div className="rate-explorer-filter">
          {isCustomized ? (
            <>
              <div className="rate-explorer-chips">
                {activeCodes.map(code => {
                  const c = CURRENCIES.find(x => x.code === code);
                  if (!c) return null;
                  return (
                    <span key={code} className="rate-explorer-chip">
                      {c.flag} {code}
                      <button
                        className="rate-explorer-chip-remove"
                        onClick={() => removeCurrency(code)}
                        aria-label={'Remove ' + code}
                      >{'\u00d7'}</button>
                    </span>
                  );
                })}
              </div>
              <div className="rate-explorer-search-wrapper">
                <input
                  type="text"
                  className="rate-explorer-search"
                  placeholder="Add currency..."
                  value={searchQuery}
                  onChange={(e) => setSearchQuery(e.target.value)}
                />
                {searchMatches.length > 0 && (
                  <div className="rate-explorer-dropdown">
                    {searchMatches.map(c => (
                      <div
                        key={c.code}
                        className="rate-explorer-dropdown-item"
                        onMouseDown={(e) => { e.preventDefault(); addCurrency(c.code); }}
                      >
                        <span className="rate-explorer-flag">{c.flag}</span>
                        <span className="rate-explorer-currency-code">{c.code}</span>
                        <span className="rate-explorer-currency-name">{c.name}</span>
                      </div>
                    ))}
                  </div>
                )}
              </div>
              <div className="rate-explorer-filter-actions">
                <button className="rate-explorer-filter-btn" onClick={resetSelection}>Reset</button>
              </div>
            </>
          ) : (
            <>
              <span className="rate-explorer-filter-label">Showing top 10 corridors</span>
              <div className="rate-explorer-search-wrapper">
                <input
                  type="text"
                  className="rate-explorer-search"
                  placeholder="Add currency..."
                  value={searchQuery}
                  onChange={(e) => setSearchQuery(e.target.value)}
                />
                {searchMatches.length > 0 && (
                  <div className="rate-explorer-dropdown">
                    {searchMatches.map(c => (
                      <div
                        key={c.code}
                        className="rate-explorer-dropdown-item"
                        onMouseDown={(e) => { e.preventDefault(); addCurrency(c.code); }}
                      >
                        <span className="rate-explorer-flag">{c.flag}</span>
                        <span className="rate-explorer-currency-code">{c.code}</span>
                        <span className="rate-explorer-currency-name">{c.name}</span>
                      </div>
                    ))}
                  </div>
                )}
              </div>
              <div className="rate-explorer-filter-actions">
                <button className="rate-explorer-filter-btn" onClick={showAll}>Show all</button>
              </div>
            </>
          )}
        </div>

        <div className="rate-explorer-table-wrapper">
          <table className="rate-explorer-table">
            <thead>
              <tr>
                <th>Destination</th>
                <th>Recipient gets</th>
                <th>Bank est.</th>
                <th>Wise</th>
                <th>Grid advantage</th>
              </tr>
            </thead>
            <tbody>
              {corridors.length === 0 && (
                <tr>
                  <td colSpan={5} className="rate-explorer-empty">
                    No currencies selected. Use the search above to add corridors.
                  </td>
                </tr>
              )}
              {corridors.map((row) => {
                const rateDec = getRateDecimals(row.midRate);
                const recDec = getReceiveDecimals(row.midRate);
                return (
                  <tr key={row.code} className="rate-explorer-row">
                    <td>
                      <div className="rate-explorer-dest">
                        <span className="rate-explorer-flag">{row.flag}</span>
                        <span className="rate-explorer-currency-code">{row.code}</span>
                        <span className="rate-explorer-currency-name">{row.name}</span>
                      </div>
                    </td>
                    <td className="rate-explorer-grid-receive">
                      <div className="rate-explorer-receive-amount">
                        {formatNumber(row.gridReceive, recDec)} {row.code}
                      </div>
                      <div className="rate-explorer-rate-subtitle">
                        Rate: {formatNumber(row.gridRate, rateDec)} (mid: {formatNumber(row.midRate, rateDec)})
                      </div>
                    </td>
                    <td className="rate-explorer-competitor">
                      {formatNumber(row.bankReceive, recDec)} {row.code}
                    </td>
                    <td className="rate-explorer-competitor">
                      {formatNumber(row.wiseReceive, recDec)} {row.code}
                    </td>
                    <td className="rate-explorer-advantage">
                      +{formatNumber(row.advantage, recDec)} ({row.advantagePct.toFixed(1)}%)
                    </td>
                  </tr>
                );
              })}
            </tbody>
          </table>
        </div>

        <div className="rate-explorer-disclaimer">
          Sample rates for illustration only. Actual rates are provided in real time via the <a href="/platform-overview/core-concepts/quote-system">Quote API</a>. Bank estimates based on typical markups of 2{'\u2013'}6% over mid-market rates.
        </div>
      </div>

      <div className="rate-explorer-historical-heading not-prose">
        <h3>Historical Comparison</h3>
        <p>Grid delivers better rates consistently, not just today.</p>
      </div>

      <div className="rate-explorer not-prose">
        <div className="rate-explorer-controls">
          <div className="rate-explorer-field">
            <label className="rate-explorer-label">Corridor</label>
            <select
              className="rate-explorer-select rate-explorer-corridor-select"
              value={historicalCorridor}
              onChange={(e) => setHistoricalCorridor(e.target.value)}
            >
              {HISTORICAL_KEYS.map((key) => (
                <option key={key} value={key}>
                  {HISTORICAL_DATA[key].label}
                </option>
              ))}
            </select>
          </div>
        </div>

        <div className="rate-explorer-table-wrapper">
          <table className="rate-explorer-table">
            <thead>
              <tr>
                <th>Week</th>
                <th>Recipient gets</th>
                <th>Bank est.</th>
                <th>Wise</th>
                <th>Grid advantage</th>
              </tr>
            </thead>
            <tbody>
              {histData && histData.weeks.map((wk) => {
                const bankRate = wk.midRate * (1 - wk.bankMarkup / 100);
                const gridReceive = amount * wk.gridRate;
                const wiseReceive = amount * wk.wiseRate;
                const bankReceive = amount * bankRate;
                const advantage = gridReceive - bankReceive;
                const advantagePct = bankReceive > 0 ? (advantage / bankReceive) * 100 : 0;
                const rateDec = getRateDecimals(wk.midRate);
                const recDec = getReceiveDecimals(wk.midRate);
                return (
                  <tr key={wk.week} className="rate-explorer-row">
                    <td className="rate-explorer-week">{wk.week}</td>
                    <td className="rate-explorer-grid-receive">
                      <div className="rate-explorer-receive-amount">
                        {formatNumber(gridReceive, recDec)} {histDest}
                      </div>
                      <div className="rate-explorer-rate-subtitle">
                        Rate: {formatNumber(wk.gridRate, rateDec)} (mid: {formatNumber(wk.midRate, rateDec)})
                      </div>
                    </td>
                    <td className="rate-explorer-competitor">
                      {formatNumber(bankReceive, recDec)} {histDest}
                    </td>
                    <td className="rate-explorer-competitor">
                      {formatNumber(wiseReceive, recDec)} {histDest}
                    </td>
                    <td className="rate-explorer-advantage">
                      +{formatNumber(advantage, recDec)} ({advantagePct.toFixed(1)}%)
                    </td>
                  </tr>
                );
              })}
            </tbody>
          </table>
        </div>

        <div className="rate-explorer-disclaimer">
          Historical data is illustrative. Past performance does not guarantee future rates.
        </div>
      </div>
    </div>
  );
};
